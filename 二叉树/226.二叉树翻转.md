[226.翻转二叉树](https://leetcode-cn.com/problems/invert-binary-tree/)
```javascript
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
// 解题思路：左右节点分别需要做什么事情
// 遍历一个节点，对于一个节点来说，需要把它的左右节点换一个位置
// 需要声明一个临时变量存储其中一个节点的值
// 将左右子节点进行更换
// 递归二叉树

// 这道题用前、后序遍历都可以做
// 前序遍历方法：
var invertTree = function(root) {
    if (root == null) {
        return null
    }
    let nodeMid = root.left;
    root.left = root.right;
    root.right = nodeMid;
    invertTree(root.left);
    invertTree(root.right);

    return root;
};

// 后序遍历方法：
var invertTree = function(root) {
    if (root == null) {
        return null
    }
    invertTree(root.left);
    invertTree(root.right);
    let nodeMid = root.left;
    root.left = root.right;
    root.right = nodeMid;
    return root;
};
```
